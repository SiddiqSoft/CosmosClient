80

83
85 CosmosEndpoint() = default;
86
89 CosmosEndpoint(const std::basic_string<char>& s)
90 {
91 this->operator=(s);
92 }

97 CosmosEndpoint& operator=(const std::basic_string<char>& s)
98 {
99 std::basic_string<char> MatchAccountEndpoint = "AccountEndpoint=";
100 std::basic_string<char> MatchAccountKey = ";AccountKey=";
101
102 // The Azure Cosmos Connection string has the following format
103 // AccountEndpoint=BaseUri;AccountKey=Key
104 // The BaseUri is fully qualified name with port
105 // The Key is the base64 encoded string representing the Cosmos key
106 if (s.starts_with(MatchAccountEndpoint)) {
107 if (auto posAccountKey = s.find(MatchAccountKey); posAccountKey != std::string::npos) {
108 // We have enough to extract both the uri and the key
109 BaseUri = s.substr(MatchAccountEndpoint.length(), posAccountKey - MatchAccountEndpoint.length());
110 EncodedKey = s.substr(posAccountKey + MatchAccountKey.length(),
111 s.length() - (posAccountKey + MatchAccountKey.length()));
112 // Make sure to strip off the trailing ; if present
113 if (EncodedKey.ends_with(";")) EncodedKey.resize(EncodedKey.length() - 1);
114 // Store the decoded key (only for std::string)
115 Key = Base64Utils::decode(EncodedKey);
116 }
117 }
118
119 return *this;
120 }
121
122
124 operator bool() const
125 {
126 return !BaseUri.empty() && !EncodedKey.empty();
127 }
128
129
131 operator std::basic_string<char>() const
132 {
133 return string();
134 }
135
136
139 const std::basic_string<char> string() const
140 {
141 return std::format("AccountEndpoint={};AccountKey={};", BaseUri, EncodedKey);
142 }
143
144
147 const auto& currentReadUri() const
148 {
149 if (!ReadableUris.empty() && CurrentReadUriId < ReadableUris.size()) return ReadableUris.at(CurrentReadUriId);
150 return BaseUri;
151 }
152
155 const auto& currentWriteUri() const
156 {
157 if (!WritableUris.empty() && CurrentWriteUriId < WritableUris.size()) return WritableUris.at(CurrentWriteUriId);
158 return BaseUri;
159 }
160
163 CosmosEndpoint& rotateReadUri()
164 {
165 if (ReadableUris.empty()) // If empty; use the baseUri
166 CurrentReadUriId = 0;
167 else if (ReadableUris.size() == 1) // Do not increment if there is only one item
168 CurrentReadUriId = 0;
169 else if (ReadableUris.size() >= 2) {
170 // We have atleast two items
171 // Increment..and..Go back to the top if we reach past the last element.
172 if (++CurrentReadUriId >= ReadableUris.size()) CurrentReadUriId = 0;
173 }
174
175 return *this;
176 }
177
180 CosmosEndpoint& rotateWriteUri()
181 {
182 if (WritableUris.empty()) // If empty, use the baseUri
183 CurrentWriteUriId = 0;
184 else if (WritableUris.size() == 1) // Do not increment if there is only one item
185 CurrentWriteUriId = 0;
186 else if (WritableUris.size() >= 2) {
187 // We have atleast two items
188 // Increment..and..Go back to the top if we reach past the last element.
189 if (++CurrentWriteUriId >= WritableUris.size()) CurrentWriteUriId = 0;
190 }
191
192 return *this;
193 }
194 };

215 struct CosmosConnection
216 {
218 enum class CurrentConnectionIdType : uint16_t
219 {
220 PrimaryConnection = 1,
221 SecondaryConnection = 2
222 };
223
225 NLOHMANN_JSON_SERIALIZE_ENUM(CurrentConnectionIdType,
226 {{CurrentConnectionIdType::PrimaryConnection, "PrimaryConnection"},
227 {CurrentConnectionIdType::SecondaryConnection, "SecondaryConnection"}});
228
230 CurrentConnectionIdType CurrentConnectionId {CurrentConnectionIdType::PrimaryConnection};
231
233 CosmosEndpoint Primary {};
234
236 CosmosEndpoint Secondary {};
237
239 CosmosConnection() = default;
240
244 CosmosConnection(const std::basic_string<char>& p, const std::basic_string<char>& s = {})
245 {
246 configure({{"connectionStrings", {p, s}}});
247 }
248
253 CosmosConnection& configure(const nlohmann::json& config)
254 {
255 if (config.contains("connectionStrings") && config.at("connectionStrings").is_array()) {
256 Primary = config.value("/connectionStrings/0"_json_pointer, "");
257 Secondary = config.value("/connectionStrings/1"_json_pointer, "");
258
259 if (!Primary) throw std::invalid_argument("Primary must be present");
260 }
261
262 // If we have readLocations then load them up for the current connection
263 if (config.contains("readableLocations")) {
264 for (auto& item : config.at("readableLocations")) {
265 if (CurrentConnectionId == CurrentConnectionIdType::SecondaryConnection)
266 Secondary.ReadableUris.push_back(item.value("databaseAccountEndpoint", ""));
267 else
268 Primary.ReadableUris.push_back(item.value("databaseAccountEndpoint", ""));
269 }
270 }
271
272 // If we have writeLocations then load them up for the current connection
273 if (config.contains("writableLocations")) {
274 for (auto& item : config.at("writableLocations")) {
275 if (CurrentConnectionId == CurrentConnectionIdType::SecondaryConnection)
276 Secondary.WritableUris.push_back(item.value("databaseAccountEndpoint", ""));
277 else
278 Primary.WritableUris.push_back(item.value("databaseAccountEndpoint", ""));
279 }
280 }
281
282 return *this;
283 }
284
285
289 const CosmosEndpoint& current() const
290 {
291 return (CurrentConnectionId == CurrentConnectionIdType::SecondaryConnection) ? std::ref(Secondary) : std::ref(Primary);
292 }

294
298 CosmosConnection& rotate(const uint16_t c = 0)
299 {
300 if (c == 0) {
301 // Swap between Primary and Secondary
302 if (CurrentConnectionId == CurrentConnectionIdType::PrimaryConnection)
303 CurrentConnectionId = CurrentConnectionIdType::SecondaryConnection;
304
305 else if (CurrentConnectionId == CurrentConnectionIdType::SecondaryConnection)
306 CurrentConnectionId = CurrentConnectionIdType::PrimaryConnection;
307 }
308 else if (c == 1) {
309 CurrentConnectionId = CurrentConnectionIdType::PrimaryConnection;
310 }
311 else if (c == 2) {
312 CurrentConnectionId = CurrentConnectionIdType::SecondaryConnection;
313 }
314
315 // If Secondary is empty; limit to Primary
316 if ((CurrentConnectionId == CurrentConnectionIdType::SecondaryConnection) && Secondary.EncodedKey.empty())
317 CurrentConnectionId = CurrentConnectionIdType::PrimaryConnection;
318
319 return *this;
320 }
321 };

456 CosmosClient& configure(const nlohmann::json& src = {}) noexcept(false)
457 {
458 if (!src.empty()) {
459 // The minimum is that the ConnectionStrings exist with at least one element; a string from the Azure portal with
460 // the Primary Connection String.
461 if (!src.contains("connectionStrings")) throw std::invalid_argument("connectionStrings missing");
462 // The current implementation requires that the Azure Cosmos service be configured with
463 // partition keys--most non-trivial implementations use Geo-spatial deployments and partition keys
464 // are required during setup.
465 if (!src.contains("partitionKeyNames")) throw std::invalid_argument("partitionKeyNames missing");
466
467 // Update our local configuration
468 m_config.update(src);
469
470 // We continue configuring..
471 // After the update we must ensure that the requirements are still met: ConnectionStrings array
472 if (!m_config["connectionStrings"].is_array()) throw std::invalid_argument("connectionStrings must be array");
473 if (m_config["connectionStrings"].size() < 1)
474 throw std::invalid_argument("connectionStrings array must contain atleast primary element");
475
476 // Update the database configuration
477 Cnxn.configure(m_config);
478
479 // Discover the regions..
480 auto [rc, regionInfo] = discoverRegions();
481 if (rc == 200 && !regionInfo.empty()) {
482 serviceSettings = regionInfo;
483 // Reconfigure/update the information such as the read location
484 Cnxn.configure(serviceSettings);
485 // Mark as updated
486 m_isConfigured = true;
487 }
488 }
489
490 return *this;
491 }

805 CosmosIterableResponseType query(const std::string& dbName,
806 const std::string& collName,
807 const std::string& pkId,
808 const std::string& queryStatement,
809 const nlohmann::json& params = {},
810 const std::string& continuationToken = {})
811 {
812 CosmosIterableResponseType ret {0xFA17, {}, {}};
813 auto ts = DateUtils::RFC7231();
814 auto count = 0;
815 nlohmann::json headers {
816 {"Authorization",
817 EncryptionUtils::CosmosToken<char>(
818 Cnxn.current().Key, "POST", "docs", std::format("dbs/{}/colls/{}", dbName, collName), ts)},
819 {"x-ms-date", ts},
820 {"x-ms-max-item-count", -1}, // -1: Let Cosmos figure out item count
821 {"x-ms-documentdb-isquery", "true"},
822 {"x-ms-version", m_config["apiVersion"]},
823 {"Content-Type", "application/query+json"}};
824
825 if (queryStatement.empty()) throw std::invalid_argument("Missing queryStatement");
826
827 if (pkId.starts_with("*")) {
828 // Special case query with partitioned data set.
829 headers["x-ms-documentdb-query-enablecrosspartition"] = "true";
830 // This is required if the client does not provide partitionkey
831 headers["x-ms-query-enable-crosspartition"] = "true";
832 }
833 else if (!pkId.empty()) {
834 // Specific partition set by client.
835 headers["x-ms-documentdb-partitionkey"] = nlohmann::json {pkId};
836 }
837
838 if (!continuationToken.empty()) {
839 headers["x-ms-continuation"] = continuationToken;
840 }
841
842 RestClient.send(ReqPost {std::format("{}dbs/{}/colls/{}/docs", Cnxn.current().currentWriteUri(), dbName, collName),
843 headers,
844 !params.is_null() && params.is_array()
845 ? nlohmann::json {{"query", queryStatement}, {"parameters", params}}
846 : nlohmann::json {{"query", queryStatement}}},
847 [&ret](const auto& req, const auto& resp) {
848 ret = {std::get<0>(resp.status()), // status code
849 resp.success() ? std::move(resp["content"]) : nlohmann::json {}, // document or empty json
850 resp["headers"].value("x-ms-continuation", "")}; // continuation token or empty
851 });
852 return ret;
853 }
